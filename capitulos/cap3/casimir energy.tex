\section{Energia de Casimir}
No espaço-tempo plano, a energia de casimir surge como consequência das condições de contorno impostas no campo quântico. Como impusemos duas condições de contorno dinstintas ao sistema \eqref{eq:kmukmu}, Um acontribuição finita da energia de Casimir pode ser calculada. Usando a equação \eqref{eq:omega} na equação \eqref{eq:modeSum}, que obtivemos no capítulo 1, a energia de vácuo do campo escalar acoplado ao éter é dada por
\begin{equation}
    E = \frac{1}{2} \left( \frac{L}{2 \pi} \right)^\ell \int d^\ell k \sum_m \sum_n \sqrt{k^2 + k_m^2 + (1+\alpha_\phi^2)k_n^2} \label{eq:modeSumFifth}
\end{equation}
Onde $k^2= k_x^2 + k_y^2$, $L^2$ é a área das placas e $\ell=2$. O parâmetro $\ell$ é chamado de regularizador de dimensão. Nós o usamos em vez de 2, porque a soma e a integral na equação \eqref{eq:modeSumFifth} são divergentes. O processo de regularização é necessário para obter um resultado finito para a energia de Casimir. Para isso usaremos a relação 
\begin{equation}
    \int f(k)d^nk = \frac{2\pi^{n/2}}{\Gamma(n/2)} \int_0^\infty f(k) k^{n-1} dk \label{eq:genSphericalCoordInt}
\end{equation}
que é a generalização da mudança de coordenadas esféricas para $n$ dimensões, onde $\Gamma(n)$ é a função gama, dada por $ \Gamma(x) = (x-1)! =  \int_0^\infty t^{x-1} e^{-t} dt $. Substituindo a equação \eqref{eq:genSphericalCoordInt} na equação \eqref{eq:modeSumFifth}, obtemos
\begin{equation}
    E = \frac{1}{2} \left( \frac{L}{2 \pi} \right)^\ell \frac{2\pi^{\ell/2}}{\Gamma(\ell/2)} \sum_m \sum_n \int_0^\infty dk k^{\ell-1} [k^2 + k_m^2 + (1+\alpha_\phi^2)k_n^2]^{-r} \label{eq:modeSumFifthSpherical}
\end{equation}
para calcular a integral em \eqref{eq:modeSumFifthSpherical} definamos $A = k_m^2 + (1+\alpha_\phi^2)k_n^2$, então a integral se torna

\begin{equation}
    I(A) =  \int_o^\infty dkk^\ell-1 (k^2 + A)^{-r}
\end{equation}

se fizermos a substituição $k=\sqrt{A}t$, obtemos

\begin{equation}
    I(A) = A^{\frac{\ell}{2}-r} \int_0^\infty dt t^{\ell-1} (1 + t^2)^{-r}
\end{equation}

Agora fazemos a substituição $t^2 = u$, então $t^{\ell-1} dt = \frac{1}{2} u^{\frac{\ell}{2}-1} du$, e a integral se torna

\begin{equation}
    I(A) = \frac{1}{2} A^{\frac{\ell}{2}-r} \int_0^\infty du u^{\frac{\ell}{2}-1} (1 + u)^{-r}
\end{equation}

usando a definição da função beta, que é dada por
\begin{equation}
    B(x,y) = \int_0^\infty du u^{x-1} (1 + u)^{-x-y} = \frac{\Gamma(x)\Gamma(y)}{\Gamma(x+y)} \label{eq:betaFuncDef}
\end{equation}
podemos escrever a integral $I(A)$ como
\begin{equation}
    I(A) = \frac{1}{2} A^{\frac{\ell}{2}-r} B\left( \frac{\ell}{2}, r - \frac{\ell}{2} \right) = \frac{1}{2} A^{\frac{\ell}{2}-r} \frac{\Gamma(\frac{\ell}{2})\Gamma(r-\frac{\ell}{2})}{\Gamma(r)} \label{eq:integralResult}
\end{equation}
Substituindo a equação \eqref{eq:integralResult} na equação \eqref{eq:modeSumFifthSpherical}, obtemos
\begin{equation}
    E = \frac{1}{4} \left( \frac{L}{2 \pi} \right)^\ell \frac{2\pi^{\ell/2}}{\Gamma(\ell/2)} \frac{\Gamma(\frac{\ell}{2})\Gamma(r-\frac{\ell}{2})}{\Gamma(r)} \sum_m \sum_n [k_m^2 + (1+\alpha_\phi^2)k_n^2]^{\frac{\ell}{2}-r} \label{eq:modeSumFifthSphericalResult}
\end{equation}
cancelando os fatores $\Gamma(\ell/2)$ e o 2 com o 4, obtemos
\begin{equation}
    E = \frac{1}{2} \left( \frac{L}{2 \pi} \right)^\ell \pi^{\ell/2} \frac{\Gamma(r-\frac{\ell}{2})}{\Gamma(r)} \sum_m \sum_n [k_m^2 + (1+\alpha_\phi^2)k_n^2]^{\frac{\ell}{2}-r} \label{eq:modeSumFifthSphericalResult2}
\end{equation}
Substituindo os valores de $k_m=\frac{m\pi}{a}$ e $k_n=\frac{2\pi}{b}(n+\beta)$, obtemos
\begin{equation}
    E = \frac{1}{2} \left( \frac{L}{2 \pi} \right)^\ell \pi^{\ell/2} \frac{\Gamma(r-\frac{\ell}{2})}{\Gamma(r)} \sum_m \sum_n \left[ \left( \frac{m\pi}{a} \right)^2 + (1+\alpha_\phi^2) \left( \frac{2\pi}{b}(n+\beta) \right)^2 \right]^{\frac{\ell}{2}-r} \label{eq:modeSumFifthSphericalResult3}
\end{equation}
Onde $r=-1/2$ é um parâmetro de regularização que nos permite calcular a integral acima e é removido tomando o limite $r \to -1/2$ no final do cálculo. Substituiremos o expoente entre colchetes por $s=r-\frac{\ell}{2}$, então obtemos
\begin{equation}
    E = \left( \frac{2}{L} \right)^{2s+1} \frac{\pi^{s + 1/2}\Gamma(s)}{2\Gamma(-1/2)}  \left(\frac{\pi}{a} \right)^{-2s} \sum_{m=0}^\infty \sum_{n=-\infty}^{\infty} \left[ \frac{a^2}{\tilde{b}^2} (n + \beta)^2 + m^2 \right]^{-s} \label{eq:modeSumFifthSphericalResult4}
\end{equation}
com
\begin{equation}
    \tilde{b} = \frac{b}{2\sqrt{1+\alpha_\phi^2}} \label{eq:tildeb}
\end{equation}

Como discutido na Ref. \cite{CarrollTam2008}, o parâmetro $\alpha_\phi$ pode ser tão grande quanto $10^{15}$ e, portanto, age como uma redução de comprimento, como na equação $\eqref{eq:tildeb}$. Em $\eqref{eq:modeSumFifthSphericalResult4}$ há duas somas. Vamos escolher a soma sobre $m$ usando a equação de Abel-Plana \eqref{eq:abelPlana} novamente como fizemos no capítulo 2. Ao considerar primeiro a substituição

\begin{equation}
    f(m) = \left[ \frac{a^2}{\tilde{b}^2} (n + \beta)^2 + m^2 \right]^{-s} = [N^2 + m^2]^{-s} \label{eq:subst}
\end{equation}
Uma vez que $z=\pm iN$ são os pontos de ramificação na última integral de $\eqref{eq:abelPlana}$ quando susbtituimos \eqref{eq:subst} em \eqref{eq:abelPlana}, de modo que obtemos 
\begin{equation}
    \sum_{m=0}^\infty f(m) = \int_0^\infty dx(N^2 + x^2)^{-s} + \frac{N^{-2s}}{2} + i \int_N^\infty dx \frac{(x^2 - N^2)^{-s}}{e^{2x\pi} - 1}. \label{eq:abelPlanaSubst}
\end{equation}
Usando a relação $(e^{2x} - 1)^{-1} = \sum_{k=1}^\infty e^{-2kx}$, podemos reescrever a segunda integral acima, temos
\begin{equation}
    \int_N^\infty dx \frac{(x^2 - N^2)^{-s}}{e^{2x\pi} - 1} = \sum_{q=1}^\infty \int_N^\infty dx (x^2 - N^2)^{-s} e^{-2q x\pi}.  \label{eq:secondIntegral}
\end{equation}
Fazendo uma mudança de variável $x=Nt$, a última integral se torna
\begin{equation}
    I_q = N^{1-2s} \int_1^\infty dt (t^2 - 1)^{-s} e^{-2q N t \pi}. \label{eq:secondIntegralSubst}
\end{equation}
Utilizando a função de Bessel modificada do segundo tipo, que é dada por
\begin{equation}
    K_\nu(z) = \frac{\sqrt{\pi}(z/2)^\nu}{\Gamma(\nu+1/2)} \int_1^\infty  e^{-zt} (t^2 - 1)^{\nu - 1/2} dt \label{eq:besselK}
\end{equation}
Identificando $\nu - 1/2 = - s$ e $z = 2\pi qN$, lembrando que $K_{-\nu} (z) = K_\nu (z)$ e fazendo algumas manipulações algébricas obtemos

\begin{equation}
    I_q = \Gamma(1-s) \pi^{-1+s} \left( \frac{q}{N} \right)^{-1/2 + s} K_{1/2-s}(2\pi qN) . \label{eq:secondIntegralBessel}
\end{equation}

Então a equação \eqref{eq:secondIntegral} se torna
\begin{equation}
    \int_N^\infty dx \frac{(x^2 - N^2)^{-s}}{e^{2x\pi} - 1} = \sum_{q=1}^\infty \left( \frac{q}{N} \right)^{-1/2 + s} \pi^{-1+s} K_{1/2-s}(2\pi qN) \Gamma(1-s). \label{eq:secondIntegralBesselSum}
\end{equation}
Substituindo \eqref{eq:subst} na equação \eqref{eq:secondIntegralBesselSum}, temos

\begin{equation}
\begin{aligned}
    E =& \left( \frac{2}{L} \right)^{2s+1} \frac{\pi^{s+1/2}\Gamma(s)}{2\Gamma(-1/2)} \left( \frac{\pi}{a} \right)^{-2s} \sum_{n=-\infty}^{\infty} \left\{ \int_{0}^{\infty} dx (N^2 + x^2)^{-s} + \frac{N^{-2s}}{2} \right. \\
     &+ \left. 2i^{1-2s} \pi^{-1+s} \Gamma(1-s) \sum_{q=1}^\infty \left( \frac{q}{N} \right)^{-1/2 + s} \pi^{-1+s} K_{1/2-s}(2\pi qN)  \right\} \label{eq:energyWithK}
\end{aligned}
\end{equation}

Dentre os termos que estão entre chaves, a integral diverge. No entanto, vamos primeiro calcular a soma em n para extrair a contribuição finita, que também está presente na integral acima. Depois disso, prosseguimos ao descartar a contribuição divergente com o processo de renormalização. Portanto, para o termo com integral, temos
\begin{equation}
     \int_{0}^{\infty} dx \sum_{n=-\infty}^{\infty} (N^2 + x^2)^{-s} = \int_0^\infty dx \left( \frac{a}{\tilde{b}} \right)^{-2s} \sum_{n=-\infty}^{\infty} \left[ (n + \beta)^{-2s} + \frac{\tilde{b}^2}{a^2}x^2 \right]^{-s}. \label{eq:abfatorados}
\end{equation}

O somatório pode ser calculado ao usar a expressão \eqref{eq:A1}, obtendo
\begin{equation}
    \begin{aligned}
     \left( \frac{a}{\tilde{b}} \right)^{-2s} dx \int_0^\infty  \sum_{n=-\infty}^{\infty} \left[ (n + \beta)^{-2s} + \frac{\tilde{b}^2}{a^2}x^2 \right]^{-s} =& \left( \frac{a}{\tilde{b}} \right)^{-2s} dx  \int_0^\infty \left[ 2\tilde{b}^{-2s} \int_0^\infty d\mu \left( \frac{\mu^2}{b^2} + \frac{x^2}{a^2} \right)^{-s} \right. \\
     &+  sen(\pi s) \frac{4\Gamma(1 - s)}{\pi^{1-s}} \left( \frac{\tilde{b}}{a}x \right)^{1-2s} \\ 
     & \quad \times \left.  \sum_{n=-1}^\infty \frac{cos(2\pi n \beta)}{n^{1/2-s}} K_{1/2-s} \left( 2\pi nx \frac{\tilde{b}}{a} \right) \right] \label{eq:eqComMu}
    \end{aligned}
\end{equation}

onde $\mu= \frac{\tilde{b}}{a}x$. A integral em $\mu$ diverge, e deve ser removida, pois ela é responsável pela contribuição divergente da energia de Casimir numa configuração em que não haveria fronteiras. A integral em x pode ser calculada usando a seguinte identidade
\begin{equation}
    \int_0^\infty dx \left( \frac{b}{a}x \right)^{1/2-s} K_{1/2 -s} \left( 2\pi n \frac{b}{a} x  \right) = \frac{1}{4} \left( \frac{a}{b}  \right) n ^{-3/2-s} \pi^{-1+s} \Gamma(1-s) \label{eq:semK}
\end{equation}

onde usamos a fórmula de reflexão de Euler

\begin{equation}
    \Gamma(z-1) \Gamma(z) = \frac{\pi}{sen(\pi z)} \label{eq:reflectionFormula}
\end{equation}

O segundo somatório em \eqref{eq:energyWithK} fornece uma expressão analítica, que é obtida através da relação \eqref{eq:A4}:

\begin{equation}
    \sum_{-\infty}^{infty} \left( \frac{a}{\tilde{b}} \right)^{-2s} (n+\beta)^{-2s} = \left( \frac{a}{\tilde{b}} \right)^{-2s} \pi^{2d-1/2} \frac{\Gamma(1/2 - s)}{\Gamma(s)} 2 \sum_{n=1}^\infty \frac{cos(2\pi n \beta)}{n^{1-2s}}. \label{eq:relA4}
\end{equation}

Ao substituir \eqref{eq:semK} e \eqref{eq:relA4} na equação \eqref{eq:energyWithK}, finalmente obetemos a forma fechada da energia de Casimir regularizada, que é dada por
\begin{equation}
    \begin{aligned}
    E^{reg} (\alpha,\beta) =& -\left( \frac{2}{L} \right)^{2s+1} \frac{\pi^s \Gamma(s)}{4} \left( \frac{\pi}{a} \right)^{-2s} \left\{ \left( \frac{a}{\tilde{b}} \right)^{1-2s} \frac{\pi^{-1+2s}}{\Gamma(s)} \Gamma(1-s) \sum_{n=1}^{\infty} \frac{cos(2\pi n \beta)}{n^{2-2s}} \right. \\ 
    &+ \left. \left( \frac{a}{\tilde{b}} \right)^{-2s} \pi^{2s-1/2} \frac{\Gamma(1/2 - s)}{\Gamma(s)} 2 \sum_{n=1}^{\infty} n^{2s-1} cos(2\pi n \beta) \right. \\
    &+ \left. 2^{1-2s} \pi^{-1+s} \Gamma(1-s) \sum_{n=-\infty}^{\infty} \sum_{q=1}^{\infty} \left( \frac{q}{N} \right)^{-1/2 -s} K_{-1/2 + s} (2Nq \pi) \right\}
    \end{aligned}
\end{equation}

Ao considerar $s \to -3/2$, removemos o parâmetro de regularização e obtemos a energia de Casimir renormalizada, que é dada por

\begin{equation}
    \begin{aligned}
        E_{reg} (\alpha, \beta) = - \frac{L^2 \pi^2}{4a^3} \left\{ \frac{3}{16} \left( \frac{a}{\pi \tilde{b}} \right)^4 \sum_{n=1}^{\infty} \frac{cos(2\pi n \beta)}{n^5} \right. +& \left. \frac{1}{2\pi^4} \left( \frac{a}{\tilde{b}} \right)^3 \sum_{n=1}^{\infty} \frac{cos(2\pi n \beta)}{n^4} \right. \\ 
        +& \left. \frac{1}{2\pi^2} \sum_{n=-\infty}^{\infty} \sum_{q=1}^{\infty} \left( \frac{N}{q} \right)^2 K_2 (2Nq\pi) \right\}
    \end{aligned}
\end{equation}
