
\chapter*{ApĂŞndice A: SomatĂłrio de $\sum f(n+\beta)$}
\addcontentsline{toc}{chapter}{ApĂŞndice A: SomatĂłrio de $\sum f(n+\beta)$}

deduziremos a fĂłrmula que Ă© usada para calcular a integral \eqref{eq:abfatorados}.
Partindo da fĂłrmula de Abel-Plana, que Ă© dada por
\begin{equation}
\begin{aligned}
\sum_{n=0}^{\infty} f(n+a)
&=
\int_{0}^{\infty} f(x)\,dx
+
\frac{f(a)}{2}
\\[4pt]
&\quad
+
i \int_{0}^{\infty} dx \,
\frac{f(a+ix)-f(a-ix)}{e^{2ix}-1}.
\end{aligned}
\tag{A1}
\label{eq:A1}
\end{equation}

O somatĂłrio pode ser dividido em dois intervalos positivo e negativo, de modo que:
\begin{equation}
\begin{aligned}
\sum_{n=-\infty}^{\infty} f(n+\beta)
&=
\sum_{n=0}^{\infty} f(n+\beta)
+
\sum_{n=0}^{-\infty} f(n+\beta)
-
f(\beta)
\\[4pt]
&=
\sum_{n=0}^{\infty} f(n+\beta)
+
\sum_{n=0}^{\infty} f(-n+\beta)
-
f(\beta).
\end{aligned}
\tag{A2}
\label{eq:A2}
\end{equation}

Aplicando os somatĂłrio sepradamente na fĂłrmula de Abel-Plana \eqref{eq:A1}, temos
\begin{equation}
\begin{aligned}
\sum_{n=0}^{\infty} \left[(n+\beta)^2+\mu^2\right]^{-s}
&=
\int_{0}^{\infty} dx \,\left[x^2+\mu^2\right]^{-s}
+
\frac{(\beta^2+\mu^2)^{-s}}{2}
\\[4pt]
&\quad
+
i\int_{0}^{\infty} dx \,
\frac{
\left[(\beta+ix)^2+\mu^2\right]^{-s}
-
\left[(\beta-ix)^2+\mu^2\right]^{-s}
}{
e^{2\pi x}-1
}.
\end{aligned}
\tag{A3}
\label{eq:A3}
\end{equation}

e
\begin{equation}
\begin{aligned}
\sum_{n=0}^{\infty} \left[(-n+\beta)^2+\mu^2\right]^{-s}
&=
\sum_{n=0}^{\infty} \left[(n+(-\beta))^2+\mu^2\right]^{-s}
\\[4pt]
&=
\int_{0}^{\infty} dx \,\left[x^2+\mu^2\right]^{-s}
+
\frac{\left[(-\beta)^2+\mu^2\right]^{-s}}{2}
\\[4pt]
&\quad
+
i\int_{0}^{\infty} dx \,
\frac{
\left[((-\beta)+ix)^2+\mu^2\right]^{-s}
-
\left[((-\beta)-ix)^2+\mu^2\right]^{-s}
}{
e^{2\pi x}-1
}
\\[4pt]
&=
\int_{0}^{\infty} dx \,\left[x^2+\mu^2\right]^{-s}
+
\frac{(\beta^2+\mu^2)^{-s}}{2}
\\[4pt]
&\quad
+
i\int_{0}^{\infty} dx \,
\frac{
\left[(-\beta+ix)^2+\mu^2\right]^{-s}
-
\left[(-\beta-ix)^2+\mu^2\right]^{-s}
}{
e^{2\pi x}-1
}.
\end{aligned}
\tag{A4}
\label{eq:A4}
\end{equation}

Ao substituir \eqref{eq:A3} e \eqref{eq:A4} na Eq.~\eqref{eq:A2}, obtemos
\begin{equation}
\begin{aligned}
\sum_{n=-\infty}^{\infty} \left[(n+\beta)^2+\mu^2\right]^{-s}
&=
\int_{0}^{\infty} dx \,\left[x^2+\mu^2\right]^{-s}
+
\frac{(\beta^2+\mu^2)^{-s}}{2}
\\[4pt]
&\quad
+
i\int_{0}^{\infty} dx \,
\frac{
\left[(\beta+ix)^2+\mu^2\right]^{-s}
-
\left[(\beta-ix)^2+\mu^2\right]^{-s}
}{
e^{2\pi x}-1
}
\\[4pt]
&\quad
+
\int_{0}^{\infty} dx \,\left[x^2+\mu^2\right]^{-s}
+
\frac{(\beta^2+\mu^2)^{-s}}{2}
\\[4pt]
&\quad
+
i\int_{0}^{\infty} dx \,
\frac{
\left[(-\beta+ix)^2+\mu^2\right]^{-s}
-
\left[(-\beta-ix)^2+\mu^2\right]^{-s}
}{
e^{2\pi x}-1
}
-
(\beta^2+\mu^2)^{-s}
\\[4pt]
&=
2\int_{0}^{\infty} dx \,\left[x^2+\mu^2\right]^{-s}
\\[4pt]
&\quad
+
\left(i^{1-2s}+(-i)^{1-2s}\right)
\int_{0}^{\infty} dx \,
\frac{
\left[(x+i\beta)^2-\mu^2\right]^{-s}
+
\left[(x-i\beta)^2-\mu^2\right]^{-s}
}{
e^{2\pi x}-1
}.
\end{aligned}
\tag{A5}
\label{eq:A5}
\end{equation}

A segunda integral de \eqref{eq:A5} pode ser resolvida usando a identidade
\begin{equation}
\left(e^{y}-1\right)^{-1}
=
\sum_{j=1}^{\infty} e^{-jy}.
\tag{A6}
\label{eq:A6}
\end{equation}

consequentemente, temos
\begin{align*}
\sum_{n=-\infty}^{\infty} \left[(n+\beta)^2+\mu^2\right]^{-s}
&=
2\int_{0}^{\infty} dx \,\left[x^2+\mu^2\right]^{-s}
\\[4pt]
&\quad
+
\left(i^{1-2s}+(-i)^{1-2s}\right)
\sum_{\ell=1}^{\infty}
\int_{0}^{\infty} dx \,
\Bigl(
\left[(x+i\beta)^2-\mu^2\right]^{-s} \\
&+\left[(x-i\beta)^2-\mu^2\right]^{-s}
\Bigr)e^{-2\ell\pi x}
\\[6pt]
&=
2\int_{0}^{\infty} dx \,\left[x^2+\mu^2\right]^{-s}
\\[4pt]
&\quad
+
\left(i^{1-2s}+(-i)^{1-2s}\right)
\sum_{\ell=1}^{\infty}
\Biggl\{
\int_{0}^{\infty} du \, e^{-2\pi\ell(u-i\beta)} \left[u^2-\mu^2\right]^{-s}
\\[4pt]
&\qquad\qquad\qquad\qquad
+
\int_{0}^{\infty} dv \, e^{-2\pi\ell(v+i\beta)} \left[v^2-\mu^2\right]^{-s}
\Biggr\}.
\end{align*}

A expressĂŁo acima pode ser reduzida a
\begin{equation}
\begin{aligned}
\sum_{n=-\infty}^{\infty} \left[(n+\beta)^2+\mu^2\right]^{-s}
&=
2\int_{0}^{\infty} dx \,\left[x^2+\mu^2\right]^{-s}
\\[4pt]
&\quad
+
\left(i^{1-2s}+(-i)^{1-2s}\right)
\sum_{\ell=1}^{\infty}
2\cos(2\pi \ell \beta)
\int_{\mu}^{\infty} du \, e^{-2\pi \ell u}\left[u^2-\mu^2\right]^{-s}
\\[6pt]
&=
\sqrt{\pi}\,\mu^{1-2s}
\frac{\Gamma\!\left(-\frac{1}{2}+s\right)}{\Gamma(s)}
\\[4pt]
&\quad
+
\left(i^{1-2s}+(-i)^{1-2s}\right)
\frac{2\Gamma(1-s)}{\pi^{\,1-s}} \\
& \times \sum_{\ell=1}^{\infty}
\left(\frac{\ell}{\mu}\right)^{-\frac{1}{2}+s}
\cos(2\pi \ell \beta)\,
K_{\frac{1}{2}-s}(2\pi \mu \ell)
\\[6pt]
&=
\sqrt{\pi}\,\mu^{1-2s}
\frac{\Gamma\!\left(-\frac{1}{2}+s\right)}{\Gamma(s)}
\\[4pt]
&\quad
+
sen(\pi s)\,
\frac{4\Gamma(1-s)}{\pi^{\,1-s}}
\sum_{\ell=1}^{\infty}
\left(\frac{\ell}{\mu}\right)^{-\frac{1}{2}+s}
\cos(2\pi \ell \beta)\,
K_{\frac{1}{2}-s}(2\pi \mu \ell).
\end{aligned}
\tag{A7}
\label{eq:A7}
\end{equation}
